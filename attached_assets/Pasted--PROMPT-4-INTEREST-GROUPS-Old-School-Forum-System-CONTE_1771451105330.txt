```typescript
// In the KEYS object, add:
FORUM_JOINED_GROUPS: "@operator_forum_joined_groups",
FORUM_USER_THREADS: "@operator_forum_user_threads",

// Interfaces
export interface JoinedGroup {
  groupId: string;
  joinedAt: string;
}

// Storage functions
export async function getJoinedGroups(): Promise<JoinedGroup[]> {
  try {
    const data = await AsyncStorage.getItem(KEYS.FORUM_JOINED_GROUPS);
    return data ? JSON.parse(data) : [];
  } catch {
    return [];
  }
}

export async function joinGroup(groupId: string): Promise<void> {
  try {
    const groups = await getJoinedGroups();
    if (!groups.find(g => g.groupId === groupId)) {
      groups.push({ groupId, joinedAt: new Date().toISOString() });
      await AsyncStorage.setItem(KEYS.FORUM_JOINED_GROUPS, JSON.stringify(groups));
    }
  } catch {}
}

export async function leaveGroup(groupId: string): Promise<void> {
  try {
    const groups = await getJoinedGroups();
    const filtered = groups.filter(g => g.groupId !== groupId);
    await AsyncStorage.setItem(KEYS.FORUM_JOINED_GROUPS, JSON.stringify(filtered));
  } catch {}
}
```

## CRITICAL STYLE RULES â€” OLD SCHOOL FORUM AESTHETIC

1. **NO rounded cards with shadows.** Use `borderWidth: 1` and `borderColor: theme.border` with sharp corners (`borderRadius: 4` max, or use `BorderRadius.sm`).
2. **Borders touching.** Thread rows, post rows, and group rows should have NO vertical gaps. The bottom border of one row IS the top border of the next (use `borderBottomWidth: 1` only).
3. **Data-dense.** Show stats everywhere: reply counts, view counts, post counts, member counts, join dates, streaks. Forums are information-rich.
4. **Monospace numbers.** Use `fontFamily: Platform.select({ ios: "Menlo", default: "monospace" })` for numeric stats (reply counts, view counts, post counts).
5. **Uppercase section headers.** All section/category headers: uppercase, letterSpacing: 2, fontSize: 11, fontWeight: "700".
6. **Muted palette.** The only color pops should be `theme.accent` (red) for hot threads, new post dots, official badges, and OP tags. Everything else uses `theme.text`, `theme.textSecondary`, and borders.
7. **Separator lines.** Use thin 1px borders to separate content, not spacing/gaps.
8. **No emojis in UI chrome.** The ğŸ“Œ and ğŸ”’ in thread titles are exceptions because they're in user content. UI elements (buttons, labels, badges) should use icons or text only.
